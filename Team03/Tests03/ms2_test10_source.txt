procedure Bedok {
  print east;
  west = 9 + east;
  y = east - 4;
  z = west + 2;
  read Yishun;
  west = 9 + x * y * z * east + west;
  while (east < 0) {
    x = z - east;
    if (east > x) then {
      x = x + 1;
      east = z - 9 + (x * y) * z; }
    else {
      y = east * y * z;
      read Yishun; }}
  west = 9 + x * y * z * east + west;
  print west; }
